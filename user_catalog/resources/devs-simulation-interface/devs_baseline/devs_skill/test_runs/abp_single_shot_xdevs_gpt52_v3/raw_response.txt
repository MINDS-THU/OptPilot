#!/usr/bin/env python3

import argparse
import sys
import json
import logging
from collections import deque
import random  # allowed (not necessarily used)
import simpy   # allowed (not used)
from xdevs.models import Atomic, Coupled, Port
from xdevs.sim import Coordinator, SimulationClock

INFINITY = float("inf")
EPS = 1e-9

# ---------------------------
# KPI JSONL output
# ---------------------------

def _t2(t: float) -> float:
    # Keep at least 2 decimals; ensure float type in JSON
    return float(f"{t:.2f}")

def emit_kpi(time_ms: float, entity: str, event: str, payload: dict):
    rec = {
        "time": _t2(time_ms),
        "entity": entity,
        "event": event,
        "payload": payload
    }
    print(json.dumps(rec), file=sys.stdout, flush=True)

# ---------------------------
# Sender (ABP stop-and-wait)
# ---------------------------

class Sender(Atomic):
    def __init__(self, name: str, parent: Coupled | None,
                 total_packets: int, sender_delay: float, timeout: float):
        super().__init__(name)
        self.parent = parent

        self.add_in_port(Port(dict, "in_ack"))
        self.add_out_port(Port(dict, "out_pkt"))

        self.total_packets = int(total_packets)
        self.sender_delay = float(sender_delay)
        self.timeout = float(timeout)

        # Time tracking (absolute time in ms)
        self._time = 0.0

        # Scheduling helpers
        self._remaining = INFINITY   # time to next internal from "now"
        self._last_sigma = INFINITY  # sigma used for next deltint time advance

        # Protocol state
        self._seq = 1
        self._bit = 0
        self._current_packet = None  # dict: {"seq_num": int, "bit": int}
        self._out_pkt = None
        self._last_send_is_retry = False
        self._is_retry_next_send = False

    def _schedule(self, phase: str, sigma: float):
        self.hold_in(phase, sigma)
        self._remaining = float(sigma)
        self._last_sigma = float(sigma)

    def initialize(self):
        self._time = 0.0
        self._seq = 1
        self._bit = 0
        self._current_packet = None
        self._out_pkt = None
        self._last_send_is_retry = False
        self._is_retry_next_send = False

        if self.total_packets <= 0:
            self._schedule("DONE", INFINITY)
            return

        # Start preparing first packet immediately at t=0
        self._current_packet = {"seq_num": self._seq, "bit": self._bit}
        self._out_pkt = dict(self._current_packet)
        self._is_retry_next_send = False

        emit_kpi(self._time, "sender", "delay_start",
                 {"type": "preparation", "duration": float(self.sender_delay)})
        self._schedule("PREPARE", self.sender_delay)

    def lambdaf(self):
        # Output only
        if self.phase == "PREPARE" and self._out_pkt is not None:
            self.output["out_pkt"].add(dict(self._out_pkt))

    def deltint(self):
        # Advance absolute time by elapsed internal delay
        if self._last_sigma < INFINITY:
            self._time += self._last_sigma

        if self.phase == "PREPARE":
            # Packet is being handed to subnet now (output occurred in lambdaf)
            if self._current_packet is None:
                # Should not happen; passivate
                self._schedule("DONE", INFINITY)
                return

            emit_kpi(self._time, "sender", "packet_sent",
                     {"seq_num": int(self._current_packet["seq_num"]),
                      "bit": int(self._current_packet["bit"]),
                      "is_retry": bool(self._last_send_is_retry)})

            # Start/restart timeout
            self._schedule("WAIT_ACK", self.timeout)

        elif self.phase == "WAIT_ACK":
            # Timeout -> retransmit same packet after preparation delay
            if self._current_packet is None:
                self._schedule("DONE", INFINITY)
                return

            self._is_retry_next_send = True
            self._last_send_is_retry = True
            self._out_pkt = dict(self._current_packet)

            emit_kpi(self._time, "sender", "delay_start",
                     {"type": "preparation", "duration": float(self.sender_delay)})
            self._schedule("PREPARE", self.sender_delay)

        else:
            self._schedule(self.phase, INFINITY)

    def deltext(self, e):
        # External transition: receive ACKs
        self._time += float(e)

        # Update remaining time for current phase (if we continue it)
        prev_phase = self.phase
        prev_remaining = self._remaining
        if prev_remaining < INFINITY:
            prev_remaining = max(0.0, prev_remaining - float(e))

        # Process incoming ACK(s)
        acks = list(self.input["in_ack"].values)
        state_changed = False

        for ack in acks:
            ack_bit = ack.get("ack_bit") if isinstance(ack, dict) else ack
            try:
                ack_bit_int = int(ack_bit)
            except Exception:
                continue

            is_valid = (prev_phase != "DONE") and (ack_bit_int == int(self._bit))

            emit_kpi(self._time, "sender", "ack_received",
                     {"ack_bit": int(ack_bit_int), "is_valid": bool(is_valid)})

            if is_valid:
                # Move to next packet or finish
                if self._seq >= self.total_packets:
                    self._out_pkt = None
                    self._current_packet = None
                    self._schedule("DONE", INFINITY)
                    state_changed = True
                    break

                # Next packet
                self._seq += 1
                self._bit = 1 - int(self._bit)
                self._current_packet = {"seq_num": self._seq, "bit": self._bit}

                self._is_retry_next_send = False
                self._last_send_is_retry = False
                self._out_pkt = dict(self._current_packet)

                emit_kpi(self._time, "sender", "delay_start",
                         {"type": "preparation", "duration": float(self.sender_delay)})
                self._schedule("PREPARE", self.sender_delay)
                state_changed = True
                break  # stop-and-wait: after valid ACK, ignore any extra this instant

        if state_changed:
            return

        # No valid ACK -> remain in current phase with reduced remaining time
        if prev_phase == "WAIT_ACK":
            self._schedule("WAIT_ACK", prev_remaining)
        elif prev_phase == "PREPARE":
            self._schedule("PREPARE", prev_remaining)
        else:
            self._schedule(prev_phase, INFINITY)

    def exit(self):
        # No extra output required
        return


# ---------------------------
# Receiver (processing delay + 1-slot buffer)
# ---------------------------

class Receiver(Atomic):
    def __init__(self, name: str, parent: Coupled | None, receiver_delay: float):
        super().__init__(name)
        self.parent = parent

        self.add_in_port(Port(dict, "in_pkt"))
        self.add_out_port(Port(dict, "out_ack"))

        self.receiver_delay = float(receiver_delay)

        self._time = 0.0
        self._remaining = INFINITY
        self._last_sigma = INFINITY

        self._current_packet = None
        self._buffered_packet = None
        self._out_ack = None

    def _schedule(self, phase: str, sigma: float):
        self.hold_in(phase, sigma)
        self._remaining = float(sigma)
        self._last_sigma = float(sigma)

    def initialize(self):
        self._time = 0.0
        self._current_packet = None
        self._buffered_packet = None
        self._out_ack = None
        self._schedule("IDLE", INFINITY)

    def lambdaf(self):
        if self.phase == "BUSY" and self._out_ack is not None:
            self.output["out_ack"].add(dict(self._out_ack))

    def deltint(self):
        if self._last_sigma < INFINITY:
            self._time += self._last_sigma

        if self.phase == "BUSY":
            if self._current_packet is not None:
                emit_kpi(self._time, "receiver", "packet_received",
                         {"seq_num": int(self._current_packet["seq_num"]),
                          "bit": int(self._current_packet["bit"])})

            # After finishing, if buffer has one packet, start it immediately
            if self._buffered_packet is not None:
                self._current_packet = self._buffered_packet
                self._buffered_packet = None
                self._out_ack = {"ack_bit": int(self._current_packet["bit"])}

                emit_kpi(self._time, "receiver", "delay_start",
                         {"type": "processing", "duration": float(self.receiver_delay)})
                self._schedule("BUSY", self.receiver_delay)
            else:
                self._current_packet = None
                self._out_ack = None
                self._schedule("IDLE", INFINITY)
        else:
            self._schedule(self.phase, INFINITY)

    def deltext(self, e):
        self._time += float(e)

        # Decrease remaining time if currently busy and we remain busy
        prev_phase = self.phase
        prev_remaining = self._remaining
        if prev_remaining < INFINITY:
            prev_remaining = max(0.0, prev_remaining - float(e))

        pkts = list(self.input["in_pkt"].values)

        # Helper to accept a packet into processing
        def start_processing(pkt: dict):
            self._current_packet = dict(pkt)
            self._out_ack = {"ack_bit": int(self._current_packet["bit"])}
            emit_kpi(self._time, "receiver", "delay_start",
                     {"type": "processing", "duration": float(self.receiver_delay)})
            self._schedule("BUSY", self.receiver_delay)

        if prev_phase == "IDLE":
            started = False
            for pkt in pkts:
                if not started:
                    start_processing(pkt)
                    started = True
                else:
                    # Now busy; buffer capacity 1
                    if self._buffered_packet is None:
                        self._buffered_packet = dict(pkt)
                    # else drop silently
            if not started:
                self._schedule("IDLE", INFINITY)
            return

        # prev_phase == BUSY (or other): keep processing; only buffer one packet
        for pkt in pkts:
            if self._buffered_packet is None:
                self._buffered_packet = dict(pkt)
            # else drop silently

        if prev_phase == "BUSY":
            self._schedule("BUSY", prev_remaining)
        else:
            self._schedule(prev_phase, INFINITY)

    def exit(self):
        return


# ---------------------------
# Subnet (deterministic noise + fixed delay, supports multiple in-flight)
# ---------------------------

class Subnet(Atomic):
    def __init__(self, name: str, parent: Coupled | None,
                 channel: str, seed: int, channel_delay: float):
        super().__init__(name)
        self.parent = parent

        self.add_in_port(Port(dict, "in_pkt"))
        self.add_out_port(Port(dict, "out_pkt"))

        self.channel = str(channel)  # "forward" or "backward"
        self.channel_delay = float(channel_delay)

        self._time = 0.0
        self._remaining = INFINITY
        self._last_sigma = INFINITY

        self._x = int(seed)
        # pending: list of (deliver_time, packet_dict)
        self._pending = []
        self._due_to_send = []

    def _schedule(self, phase: str, sigma: float):
        self.hold_in(phase, sigma)
        self._remaining = float(sigma)
        self._last_sigma = float(sigma)

    def _recompute_next(self):
        if not self._pending:
            self._due_to_send = []
            self._schedule("IDLE", INFINITY)
            return

        next_t = min(t for (t, _) in self._pending)
        sigma = max(0.0, next_t - self._time)

        # Determine which will be due at the next internal event time
        t_event = self._time + sigma
        due = [pkt for (t, pkt) in self._pending if abs(t - t_event) <= EPS]
        self._due_to_send = due
        self._schedule("SCHEDULED", sigma)

    def initialize(self):
        self._time = 0.0
        self._pending = []
        self._due_to_send = []
        # x initialized to seed exactly
        # passivate
        self._schedule("IDLE", INFINITY)

    def lambdaf(self):
        # Send all due packets at this internal time
        for pkt in self._due_to_send:
            self.output["out_pkt"].add(dict(pkt))

    def deltint(self):
        if self._last_sigma < INFINITY:
            self._time += self._last_sigma

        if self.phase in ("SCHEDULED",):
            # Remove all packets whose delivery time == now
            new_pending = []
            for (t, pkt) in self._pending:
                if abs(t - self._time) <= EPS:
                    continue
                new_pending.append((t, pkt))
            self._pending = new_pending
            self._due_to_send = []
            self._recompute_next()
        else:
            self._schedule(self.phase, INFINITY)

    def deltext(self, e):
        self._time += float(e)

        # Process arrivals: decide fate immediately
        for pkt in list(self.input["in_pkt"].values):
            x_new = (17 * self._x + 11) % 100
            self._x = x_new
            behavior = "drop" if x_new < 10 else "pass"

            emit_kpi(self._time, "subnet", "packet_get",
                     {"behavior": behavior, "channel": self.channel, "noise_value": int(x_new)})

            if behavior == "pass":
                deliver_time = self._time + self.channel_delay
                self._pending.append((deliver_time, dict(pkt)))

        # Recompute schedule based on (possibly) new earlier deliveries
        self._recompute_next()

    def exit(self):
        return


# ---------------------------
# Coupled System
# ---------------------------

class System(Coupled):
    def __init__(self, name: str, parent: Coupled | None,
                 total_packets: int, seed: int,
                 timeout: float, sender_delay: float, receiver_delay: float, channel_delay: float):
        super().__init__(name)
        self.parent = parent

        sender = Sender("sender", self, total_packets=total_packets,
                        sender_delay=sender_delay, timeout=timeout)
        receiver = Receiver("receiver", self, receiver_delay=receiver_delay)

        subnet_fwd = Subnet("subnet1", self, channel="forward", seed=seed, channel_delay=channel_delay)
        subnet_bwd = Subnet("subnet2", self, channel="backward", seed=seed, channel_delay=channel_delay)

        self.add_component(sender)
        self.add_component(receiver)
        self.add_component(subnet_fwd)
        self.add_component(subnet_bwd)

        # Sender -> forward subnet -> Receiver
        self.add_coupling(sender.output["out_pkt"], subnet_fwd.input["in_pkt"])
        self.add_coupling(subnet_fwd.output["out_pkt"], receiver.input["in_pkt"])

        # Receiver -> backward subnet -> Sender
        self.add_coupling(receiver.output["out_ack"], subnet_bwd.input["in_pkt"])
        self.add_coupling(subnet_bwd.output["out_pkt"], sender.input["in_ack"])


# ---------------------------
# Entry point
# ---------------------------

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--total_packets", type=int, required=True)
    parser.add_argument("--seed", type=int, default=42)
    parser.add_argument("--timeout", type=int, default=20)
    parser.add_argument("--sender_delay", type=int, default=10)
    parser.add_argument("--receiver_delay", type=int, default=10)
    parser.add_argument("--channel_delay", type=int, default=3)
    parser.add_argument("--simulate_time", type=int, default=1000)
    args = parser.parse_args()

    logging.basicConfig(stream=sys.stderr, level=logging.WARNING,
                        format="%(asctime)s %(levelname)s %(message)s")

    root = System(
        name="system",
        parent=None,
        total_packets=args.total_packets,
        seed=args.seed,
        timeout=float(args.timeout),
        sender_delay=float(args.sender_delay),
        receiver_delay=float(args.receiver_delay),
        channel_delay=float(args.channel_delay),
    )

    coord = Coordinator(root, clock=SimulationClock(0))
    coord.initialize()
    coord.simulate_time(float(args.simulate_time))


if __name__ == "__main__":
    main()
